CREATE SCHEMA devops4 AUTHORIZATION student;

CREATE TABLE devops4.students (
    id SERIAL PRIMARY KEY,
    name VARCHAR(50),
    age INT
);

CREATE TABLE devops4.courses (
    id SERIAL PRIMARY KEY,
    title VARCHAR(100),
    credits INT
);

CREATE TABLE devops4.enrollments (
    student_id INT REFERENCES devops4.students(id),
    course_id INT REFERENCES devops4.courses(id),
    enrolled_on DATE DEFAULT CURRENT_DATE
);
